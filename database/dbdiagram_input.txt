Table Utente {
  id_utente int [pk, increment]
  nome string
  cognome string
  telefono string
  email string
  ruolo Ruolo
  supervisore bool 
}

Table UtenteRistorante {
  id_utente int [not null,unique]
  id_ristorante int [not null, unique]
  is_admin bool
}

Table Ristorante {
  id_ristorante int [pk, increment]
  nome string
  indirizzo string
  telefono string
  sito_web string
  foto_path string
}

Table Categoria {
  id_categoria int [pk, increment]
  id_ristorante int [not null, unique]
  nome string
}

Table Elemento {
  id_elemento int [pk, increment]
  id_categoria int [not null, unique]
  nome string
  descrizione string
  prezzo double
  ingredienti string[]
}

Table Conto {
  id_conto int [pk, increment]
  id_ristorante int [not null, unique]
  codice_tavolo string [not null]
  data date
}

Table Ordinazione {
  id_ordinazione int [pk, increment]
  id_conto int [not null, unique]
  timestamp datetime
  date date
  evaso_da int [not null, unique]
  evaso bool
}

Table ElementoConQuantita {
  id_elemento int [not null, unique]
  id_ordinazione int [not null, unique]
  quantita int
}

Table Allergene {
  id_elemento int [not null]
  nome string [pk,  not null]
}

Enum Ruolo {
  ADDETTO_CUCINA
  CAMERIERE
  ADMIN
}
